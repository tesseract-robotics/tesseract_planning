278  bool isToleranced() const
279  {
280  // Check if they are empty
281  if (lower_tolerance.size() == 0 || upper_tolerance.size() == 0)
282  return false;
283 
284  // Check if they are the same
285  static auto max_diff = static_cast<double>(std::numeric_limits<float>::epsilon());
286 
287  if ((lower_tolerance.array() > max_diff).any())
288  throw std::runtime_error("CartesianWaypoint: lower tolerance was provided but must be <= 0,");
289 
290  if ((upper_tolerance.array() < -max_diff).any())
291  throw std::runtime_error("CartesianWaypoint: upper tolerance was provided but must be >= 0,");
292 
293  return !tesseract_common::almostEqualRelativeAndAbs(lower_tolerance, upper_tolerance, max_diff);
