131template <typename ProfileType>
132std::shared_ptr<const ProfileType> getProfile(const std::string& ns,
133 const std::string& profile,
134 const ProfileDictionary& profile_dictionary,
135 std::shared_ptr<const ProfileType> default_profile = nullptr)
136{
137 if (profile_dictionary.hasProfile<ProfileType>(ns, profile))
138 return profile_dictionary.getProfile<ProfileType>(ns, profile);
139
140 CONSOLE_BRIDGE_logDebug("Profile '%s' was not found in namespace '%s' for type '%s'. Using default if available. "
141 "Available "
142 "profiles:",
143 profile.c_str(),
144 ns.c_str(),
145 typeid(ProfileType).name());
146
147 if (profile_dictionary.hasProfileEntry<ProfileType>(ns))
148 {
149 for (const auto& pair : profile_dictionary.getProfileEntry<ProfileType>(ns))
150 {
151 CONSOLE_BRIDGE_logDebug("%s", pair.first.c_str());
152 }
153 }
154
155 return default_profile;
156}
