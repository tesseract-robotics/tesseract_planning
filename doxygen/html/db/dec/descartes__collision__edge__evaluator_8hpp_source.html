71 template <typename FloatType>
72 std::pair<bool, FloatType>
73 DescartesCollisionEdgeEvaluator<FloatType>::evaluate(const descartes_light::State<FloatType>& start,
74  const descartes_light::State<FloatType>& end) const
75 {
76  assert(start.values.rows() == end.values.rows());
77 
78  // Happens in two phases:
79  // 1. Compute the transform of all objects
80  tesseract_common::TrajArray segment(2, start.values.rows());
81  for (Eigen::Index i = 0; i < start.values.rows(); ++i)
82  {
83  segment(0, i) = start[i];
84  segment(1, i) = end[i];
85  }
86 
87  std::vector<tesseract_collision::ContactResultMap> discrete_results;
88  std::vector<tesseract_collision::ContactResultMap> continuous_results;
89  bool discrete_in_contact = discreteCollisionCheck(discrete_results, segment, allow_collision_);
90  bool continuous_in_contact = continuousCollisionCheck(continuous_results, segment, allow_collision_);
91 
92  if (!discrete_in_contact && !continuous_in_contact)
93  return std::make_pair(true, 0);
94 
95  // TODO: Update this to consider link pairs
96  auto collision_safety_margin_ =
97  static_cast<FloatType>(collision_check_config_.collision_margin_data.getMaxCollisionMargin());
98 
99  if (!discrete_in_contact && continuous_in_contact && allow_collision_)
100  return std::make_pair(true, collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance);
101 
102  if (discrete_in_contact && !continuous_in_contact && allow_collision_)
103  return std::make_pair(true, collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance);
104 
105  if (discrete_in_contact && continuous_in_contact && allow_collision_)
106  {
107  double d = collision_safety_margin_ - discrete_results.begin()->begin()->second[0].distance;
108  double c = collision_safety_margin_ - continuous_results.begin()->begin()->second[0].distance;
109  return std::make_pair(true, std::max(d, c));
110  }
111 
112  return std::make_pair(false, 0);
113 }

121 template <typename FloatType>
122 bool DescartesCollisionEdgeEvaluator<FloatType>::continuousCollisionCheck(
123  std::vector<tesseract_collision::ContactResultMap>& results,
124  const tesseract_common::TrajArray& segment,
125  bool find_best) const
126 {
127  // It was time using chronos time elapsed and it was faster to cache the contact manager
128  unsigned long int hash = std::hash<std::thread::id>{}(std::this_thread::get_id());
129  tesseract_collision::ContinuousContactManager::Ptr cm;
130  tesseract_environment::StateSolver::Ptr ss;
131  mutex_.lock();
132  auto it = continuous_contact_managers_.find(hash);
133  if (it == continuous_contact_managers_.end())
134  {
135  cm = continuous_contact_manager_->clone();
136  continuous_contact_managers_[hash] = cm;
137 
138  ss = state_solver_->clone();
139  state_solver_managers_[hash] = ss;
140  }
141  else
142  {
143  cm = it->second;
144  ss = state_solver_managers_[hash];
145  }
146  mutex_.unlock();
147  tesseract_collision::CollisionCheckConfig config = collision_check_config_;
148  if (config.type == tesseract_collision::CollisionEvaluatorType::LVS_DISCRETE ||
149  config.type == tesseract_collision::CollisionEvaluatorType::LVS_CONTINUOUS)
150  config.type = tesseract_collision::CollisionEvaluatorType::LVS_CONTINUOUS;
151  else
152  config.type = tesseract_collision::CollisionEvaluatorType::CONTINUOUS;
153  config.contact_request.type =
154  (find_best) ? tesseract_collision::ContactTestType::CLOSEST : tesseract_collision::ContactTestType::FIRST;
155 
156  return tesseract_environment::checkTrajectory(results, *cm, *ss, joint_names_, segment, config);
157 }
