139 getRobotRedundancy(const Eigen::Ref<const Eigen::Matrix<FloatType, Eigen::Dynamic, 1>>& joint_values,
140  const Eigen::Ref<const Eigen::Vector2i>& sign_correction = Eigen::Vector2i::Ones())
141 {
142  Eigen::Vector3i redundancy;
143 
144  // The joints on a 6 dof robot with redundancy capability are axis 1, 5, and 6.
145  int red_1 = int(joint_values(0) / M_PI);
146  int red_2 = int(sign_correction[1] * joint_values(4) / M_PI);
147  int red_3 = int(joint_values(5) / M_PI);
148 
149  redundancy << red_1, red_2, red_3;
