79 {

82 
83  // Get the pose at tool0
84  Eigen::Isometry3d pose = state.at(base_link).inverse() * state.at(tcp_frame);
85 
86  // Get the base rotated by joint 1
87  Eigen::Isometry3d prime_pose(
88  Eigen::AngleAxisd(static_cast<double>(joint_values.tail(6)(0)), Eigen::Vector3d::UnitZ()));
89 
90  // Transform tool0 pose into new frame
91  Eigen::Isometry3d pose_prime = prime_pose.inverse() * pose;
92 
93  // If pose_prime.x is greater than and equal to zero then it is in the forward configuration, otherwise
94  // in the backward configuration.
95 
96  std::array<std::string, 3> config;
97 
98  // Wrist Flip or Not Flip
99  if ((sign_correction[1] * joint_values(4)) >= 0)
100  config[0] = "F";
101  else
102  config[0] = "N";
103 
104  // Elbow Up or Down
105  if ((sign_correction[0] * joint_values(2)) < M_PI / 2)
106  config[1] = "U";
107  else
108  config[1] = "D";
109 
110  // Robot Front or Back
111  if (pose_prime.translation().x() >= 0)
112  config[2] = "T";
113  else
114  config[2] = "B";
115

117  return RobotConfig::FUT;
